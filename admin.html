/* ============================== ตั้งค่า ============================== */ // ชื่อชีตที่เก็บข้อมูล // ถ้าชีตของคุณชื่อ Sheet1 ให้เปลี่ยนเป็น "Sheet1" const SHEET_NAME = "data"; // key ต้องตรงกับ ADMIN_KEY ในหน้า admin.html const ADMIN_KEY = "123456"; /* ============================== doGet หลัก ============================== */ function doGet(e) { const action = e.parameter.action || ""; const key = e.parameter.key || ""; const callback = e.parameter.callback || ""; let result = {}; try { if (key !== ADMIN_KEY) { result = { ok: false, error: "KEY ไม่ถูกต้อง" }; } else if (action === "list") { result = listData(); } else if (action === "clear") { result = clearData(); } else { result = { ok: false, error: "action ไม่ถูกต้อง" }; } } catch (err) { result = { ok: false, error: String(err) }; } return outputResult(result, callback); } /* ============================== ส่งออก JSON / JSONP ============================== */ function outputResult(result, callback) { const json = JSON.stringify(result); // ถ้ามี callback ให้ตอบกลับแบบ JSONP if (callback) { return ContentService .createTextOutput(callback + "(" + json + ");") .setMimeType(ContentService.MimeType.JAVASCRIPT); } // ถ้าไม่มี callback ให้ตอบกลับแบบ JSON ปกติ return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } /* ============================== โหลดข้อมูลทั้งหมด ============================== */ function listData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { return { ok: false, error: "ไม่พบชีตชื่อ " + SHEET_NAME }; } const values = sheet.getDataRange().getValues(); if (values.length <= 1) { return { ok: true, rows: [] }; } const headers = values[0]; const rows = []; for (let i = 1; i < values.length; i++) { const rowObj = {}; for (let j = 0; j < headers.length; j++) { rowObj[String(headers[j])] = values[i][j]; } rows.push(rowObj); } // แสดงข้อมูลล่าสุดขึ้นก่อน rows.reverse(); return { ok: true, rows: rows }; } /* ============================== ล้างข้อมูลทั้งหมด เหลือหัวตารางไว้ ============================== */ function clearData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { return { ok: false, error: "ไม่พบชีตชื่อ " + SHEET_NAME }; } const lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.deleteRows(2, lastRow - 1); } return { ok: true, message: "ล้างข้อมูลทั้งหมดแล้ว" }; }
